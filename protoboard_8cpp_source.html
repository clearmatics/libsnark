50 void Protoboard::setValuesAsBitArray(
51  const VariableArray &varArray, const size_t srcValue)
52 {
53  GADGETLIB_ASSERT(
54  varArray.size() >= Log2ceil(srcValue),
55  GADGETLIB2_FMT(
56  "Variable array of size %u too small to hold value %u. Array must "

60  srcValue,
